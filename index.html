            { text: "千淘万漉虽辛苦，吹尽狂沙始到金。", author: "刘禹锡" },

        ];

        

        // DOM元素

        const quoteElement = document.getElementById('quote');

        const authorElement = document.getElementById('author');

        const newQuoteButton = document.getElementById('newQuote');

        const shareQuoteButton = document.getElementById('shareQuote');

        

        // 显示随机名言

        function showRandomQuote() {

            const randomIndex = Math.floor(Math.random() * quotes.length);

            const randomQuote = quotes[randomIndex];

            

            quoteElement.textContent = randomQuote.text;

            authorElement.textContent = `— ${randomQuote.author}`;

            

            // 添加淡入动画

            quoteElement.style.opacity = 0;

            authorElement.style.opacity = 0;

            

            setTimeout(() => {

                quoteElement.style.opacity = 1;

                authorElement.style.opacity = 1;

            }, 300);

        }

        

        // 分享功能

        function shareQuote() {

            const currentQuote = quoteElement.textContent;

            const currentAuthor = authorElement.textContent;

            

            if (navigator.share) {

                navigator.share({

                    title: '智慧语录',

                    text: `${currentQuote} ${currentAuthor}`,

                    url: window.location.href

                })

                .catch(error => console.log('分享出错:', error));

            } else {

                // 备用方案 - 复制到剪贴板

                const textToCopy = `${currentQuote} ${currentAuthor}\n\n来自: ${window.location.href}`;

                

                navigator.clipboard.writeText(textToCopy)

                    .then(() => {

                        alert('名言已复制到剪贴板！');

                    })

                    .catch(err => {

                        console.error('复制失败:', err);

                        alert('复制失败，请手动复制文本。');

                    });

            }

        }

        

        // 事件监听

        newQuoteButton.addEventListener('click', showRandomQuote);

        shareQuoteButton.addEventListener('click', shareQuote);

        

        // 页面加载时显示随机名言

        document.addEventListener('DOMContentLoaded', showRandomQuote);

        

        // 添加页面刷新时也显示随机名言

        window.onload = showRandomQuote;
